lc(){
    case "$1" in
        [A-Z])
        n=$(printf "%d" "'$1")
        n=$((n+32))
        printf \\$(printf "%o" "$n")
        ;;
        *)
        printf "%s" "$1"
        ;;
    esac
}
read word
for((i=0;i<${#word};i++))
do
    ch="${word:$i:1}"
    lc "$ch"
done
